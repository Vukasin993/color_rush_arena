      console.log('ğŸ”„ Auth state changed:', user?.uid || 'No user');
      if (user) {

AFTER:
    onAuthStateChanged(this.auth, async (user) => {
      console.log('ğŸ”„ Auth state changed:', user?.uid || 'No user');
      if (user) {
        console.log('âœ… User is authenticated with UID:', user.uid);
        
        // Get user profile to fetch username
        const userProfile = await this.getUserFromStorage();
        
        // Set Sentry user context with Firebase UID and username
        Sentry.setUser({
          id: user.uid,
          username: userProfile?.username || 'Unknown',
          email: user.email || undefined,
        });
        
        Sentry.addBreadcrumb({
          category: 'auth',
          message: 'User authenticated',
          level: 'info',
          data: { uid: user.uid, username: userProfile?.username },
        });
      } else {
        console.log('âŒ No authenticated user found');
        // Clear Sentry user context
        Sentry.setUser(null);
      }
    });
